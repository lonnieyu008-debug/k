# 3.1 Pre-processing data and store new data into CSV file
import pandas as pd
dataframe = pd.read_csv("SpeedDating.csv", header=None)
dataframe.columns = variable_names[1:]

dataframe = dataframe.dropna()
print(len(dataframe))

dataframe['gender'] = dataframe['gender'].apply(lambda x: str(x).replace("b'", "").replace("'", "").strip())
dataframe['gender'] = dataframe['gender'].map({'male':1,'female':0})

dataframe['samerace'] = dataframe['samerace'].apply(lambda x: int(str(x).replace("b'", "").replace("'", "")))
dataframe['match'] = dataframe['match'].apply(lambda x: int(str(x).replace("b'", "").replace("'", "")))

dataframe.to_csv("dataforanalysis.csv", index=False)

from sklearn.tree import DecisionTreeClassifier
from sklearn.metrics import accuracy_score
from sklearn.model_selection import train_test_split 
#3.3 partition the dataframe
train_df, test_df = train_test_split(dataframe, test_size=0.2, random_state=42)

#3.3 predict matching results
X_train = train_df.drop('match', axis=1)
y_train = train_df['match']
X_test = test_df.drop('match', axis=1)
y_test = test_df['match']
def train_and_predict(X_train, y_train, X_test):
    model = DecisionTreeClassifier(random_state=42)
    model.fit(X_train, y_train)
    return model, model.predict(X_test)
model, y_predict = train_and_predict(X_train, y_train, X_test)

#3.3 evaluate the prediction results
from sklearn.metrics import classification_report, ConfusionMatrixDisplay
print("Classification Report:\n\n", classification_report(y_test, y_predict))
ConfusionMatrixDisplay.from_estimator(model, X_test, y_test)
plt.title("Confusion Matrix")
plt.show()
